267  public final ResourceBundle newBundle( final String p_basename, final Locale p_locale, final String p_format, final ClassLoader p_loader,
268  final boolean p_reload
269  ) throws IllegalAccessException, InstantiationException, IOException
270  {
271  InputStream l_stream = null;
272  final String l_resource = this.toResourceName( this.toBundleName( p_basename, p_locale ), "properties" );
273 
274  if ( !p_reload )
275  l_stream = p_loader.getResourceAsStream( l_resource );
276  else
277  {
278 
279  final URL l_url = p_loader.getResource( l_resource );
280  if ( l_url == null )
281  return null;
282 
283  final URLConnection l_connection = l_url.openConnection();
284  if ( l_connection == null )
285  return null;
286 
287  l_connection.setUseCaches( false );
288  l_stream = l_connection.getInputStream();
289  }
290 
291  try
292  {
293  return new PropertyResourceBundle( new InputStreamReader( l_stream, "UTF-8" ) );
294 
295  }
296  catch ( final Exception l_exception )
297  {
298  return null;
299  }
300  finally
301  {
302  l_stream.close();
303  }
304  }
305  }
