118  @SuppressWarnings( "unchecked" )
119  public static IGrammarElement terminalvalue( final String p_value )
120  {
121  // a terminal-string starts and ends always with ', so we do
122  // not need to parse th string if this condition fits
123  if ( ( p_value.startsWith( "'" ) ) && ( p_value.endsWith( "'" ) ) )
124  return new CGrammarTerminalValue<>( p_value );
125 
126  // try to compile string as regular expression pattern
127  try
128  {
129  return (IGrammarElement) new CASTVisitorPCRE().visit(
130  new PCREParser(
131  new CommonTokenStream(
132  new PCRELexer(
133  new ANTLRInputStream( new ByteArrayInputStream( Pattern.compile( p_value ).pattern().getBytes( "UTF-8" ) ) )
134  )
135  )
136  ).parse()
137  );
138  }
139  catch ( final PatternSyntaxException | IOException l_exception )
140  {
141  }
142 
143  // if any fails, return it directly
144  return new CGrammarTerminalValue<>( p_value );
145  }

154  public static String cleanComment( final String p_comment, final Set<String> p_docuclean )
155  {
156  if ( p_comment == null )
157  return null;
158 
159  // remove CR, LF and tab
160  final CStringReplace l_documentation = new CStringReplace( p_comment ).replaceAll( "(\\t|\\n)+", " " ).replace( "\r", "" );
161  p_docuclean.forEach( i -> l_documentation.replaceAll( i, "" ) );
162 
163  return l_documentation.replaceAll( "\\*", "" ).replaceAll( "\\/", "" ).get().trim();
