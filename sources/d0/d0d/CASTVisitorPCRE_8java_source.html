60  @Override
61  public final Object visitAtom( final PCREParser.AtomContext p_context )
62  {
63  // character class like [a-z]
64  if ( p_context.character_class() != null )
65  return this.visitCharacter_class( p_context.character_class() );
66 
67  // dot replaced with a fixed terminal but in the visitor rule above,
68  // because other characters can be also returned
69  if ( p_context.Dot() != null )
70  return p_context.Dot().getText();
71 
72  // string / character definition
73  if ( p_context.literal() != null )
74  return p_context.literal().getText();
75 
76  // defines ^
77  if ( p_context.Caret() != null )
78  return p_context.getText();
79 
80  // defines choice a | b | c
81  if ( p_context.capture() != null )
82  return this.visitCapture( p_context.capture() );
83 
84  return null;

100  @Override
101  public final Object visitExpr( final PCREParser.ExprContext p_context )
102  {
103  // iterate over elements
104  final List<Pair<?, String>> l_pairs = p_context.element().stream()
105  // pair with grammar element or string and cardinality
106  .map( i -> (Pair<?, String>) this.visitElement( i ) )
107  .filter( i -> i != null )
108  .collect( Collectors.toList() );
109 
110  // get pairs and check if a negation "~" is followed by
111  // a grammar symbol so concat this first
112  final List<Pair<?, String>> l_clean = SequenceM.range( 0, l_pairs.size() )
113  .sliding( 2, 2 )
114  .flatMap( i -> l_pairs.get( i.get( 0 ) ).getLeft().equals( "~" )
115  ? Stream.of( new ImmutablePair<>(
116  new CGrammarNegation( (IGrammarElement) l_pairs.get( i.get( 1 ) ).getLeft() ),
117  l_pairs.get( i.get( 1 ) ).getRight()
118  )
119  )
120  : i.get( 1 ) != null
121  ? Stream.of( l_pairs.get( i.get( 0 ) ), l_pairs.get( i.get( 1 ) ) )
122  : Stream.of( l_pairs.get( i.get( 0 ) ) )
123  )
124  .collect( Collectors.toList() );
125 
126  // build sequences and implode strings
127  return CCommon.sequence(
128  this.implode( l_clean ).stream()
129  .map( i -> CCommon.cardinality(
130  i.getRight(),
131  i.getLeft() instanceof String
132 
133  // terminal-value as string
134  ? this.terminalvalue( i.getLeft().toString() )
135 
136  // native grammar element
137  : (IGrammarElement) i.getLeft()
138  )
139  )
140  .collect( Collectors.toList() )
141  );
142  }

170  {

185  private List<Pair<?, String>> implode( final List<Pair<?, String>> p_list )
186  {
187  // search string within the list
188  final int l_start = this.filter( 0, p_list );
189  if ( l_start < 0 )
190  return p_list;
191 
192  final int l_end = this.filter( l_start + 1, p_list );
193  if ( l_end < 0 )
194  return p_list;
195 
196  return this.implode(
197  Stream.concat(
198  Stream.of(
199  new ImmutablePair<>(
200  StringUtils.join( p_list.subList( l_start, l_end + 1 ).stream().map( i -> i.getLeft().toString() ).collect( Collectors.toList() ), "" ),
201  p_list.get( l_end ).getRight()
202  )
203  ),
204  CASTVisitorPCRE.this.implode( p_list.subList( l_end + 1, p_list.size() ) ).stream()
205  ).collect( Collectors.toList() )
206  );
207  }
208 
216  private int filter( final int p_start, final List<Pair<?, String>> p_list )
217  {
218  return IntStream.range( p_start, p_list.size() )
219  .boxed()
220  .filter( i -> ( p_list.get( i ).getLeft() instanceof String ) || ( !p_list.get( i ).getRight().isEmpty() ) )
221  .findFirst()
222  .orElse( -1 );
223  }
