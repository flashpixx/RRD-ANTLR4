200  @Override
201  protected final String choice( final IGrammarChoice p_element )
202  {
203  final String l_child = StringUtils.join(
204  IntStream
205  .range( 0, p_element.get().size() )
206  .boxed()
207  .map( i -> this.map( p_element.get().get( i ) ) )
208  .filter( Objects::nonNull )
209  .collect( Collectors.toList() ),
210  ", "
211  );
212 
213  return p_element.get().size() == 1 ? l_child : MessageFormat.format( "Choice({0}, {1})", 0, l_child );
214  }
215 
216  @Override
217  protected final String sequence( final IGrammarCollection p_element )
218  {
219  final String l_child = StringUtils.join(
220  p_element.get().stream()

222  .filter( Objects::nonNull )
223  .collect( Collectors.toList() ),
224  ", "
225  );
226
