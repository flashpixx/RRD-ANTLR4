81  public final void postprocess( final Path p_output ) throws IOException, URISyntaxException
82  {
83  this.copy( "index.tex", p_output );
84 
85  // replace content
86  this.replace(
87  new File( p_output.toString(), "/index.tex" ),
88 
89  // set title
90  "-grammartitle-", CCommon.languagestring( this, "section", m_grammar.id() ),
91 
92  // set grammar documentation
93  "-grammardocumentation-", m_grammar.documentation(),
94 
95  // set text rules
96  "-rules-", StringUtils.join(
97  m_rules.rowMap().entrySet().stream().sorted( ( n, m ) -> n.getKey().compareToIgnoreCase( m.getKey() ) )
98  .map( i -> MessageFormat.format(
99  "\\subsection*'{'{0}'}'\n"
100  + "\\begin'{'grammar'}'"
101  + "\n{1}\n"
102  + "\\end'{'grammar'}'",
103  CCommon.languagestring( this, "subsectiongrammar", i.getKey() ),
104  StringUtils.join(
105  i.getValue().entrySet().stream()
106  .sorted( ( n, m ) -> n.getKey().compareToIgnoreCase( m.getKey() ) )
107  .map( Map.Entry::getValue )
108  .collect( Collectors.toList() ),
109  "\n"
110  ).trim()
111  )
112  )
113  .collect( Collectors.toList() ),
114  "\n\n"
115  )
116  );
117  }

144  @Override
145  protected final String cardinality( final IGrammarElement.ECardinality p_cardinality, final String p_element )
146  {
147  switch ( p_cardinality )
148  {
149  case OPTIONAL:
150  return MessageFormat.format( "{0}{1}", p_element, "?" ).trim();
151 
152  case ZEROORMORE:
153  return MessageFormat.format( "{0}{1}", p_element, "*" ).trim();
154 
155  case ONEORMORE:
156  return MessageFormat.format( "{0}{1}", p_element, "+" ).trim();
157 
158  default:
159  return p_element;
160  }
161  }

213  private static String escapelatex( final String p_string )
214  {
215  // ", ~ and ^ are replaced by the syntax package itself
216  return CLaTeXSyntax.removequotes( p_string )
217  .replaceAll( "\\\\", "\\\\textbackslash " )
218  .replaceAll( "([\\$\\#&%_])", "\\\\$1" )
219  .replaceAll( "<", "\\\\textless " )
220  .replaceAll( ">", "\\\\textgreater " )
221  .replaceAll( "\\{", "\\\\{" )
222  .replaceAll( "\\}", "\\\\}" )
223  .replaceAll( "\\|", "\\\\vline\\\\xspace" );
224  }
