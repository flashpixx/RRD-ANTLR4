83  @Override
84  public final IGrammarElement visitDelegateGrammar( final ANTLRv4Parser.DelegateGrammarContext p_context )
85  {
86  p_context.id().stream().map( i -> (IGrammarSimpleElement<String>) this.visitId( i ) ).forEach( i -> m_imports.add( i ) );
87  return this.visitChildren( p_context );
88  }

197  @Override
198  public final IGrammarElement visitElement( final ANTLRv4Parser.ElementContext p_context )
199  {
200  if ( p_context.labeledElement() != null )
201  return CCommon.cardinality(
202  p_context.ebnfSuffix() != null
203  ? p_context.ebnfSuffix().getText()
204  : "",
205  this.visitLabeledElement( p_context.labeledElement() )
206  );
207 
208 
209  if ( p_context.atom() != null )
210  return CCommon.cardinality(
211  p_context.ebnfSuffix() != null
212  ? p_context.ebnfSuffix().getText()
213  : "",
214  this.visitAtom( p_context.atom() )
215  );
216 
217  if ( p_context.ebnf() != null )
218  return this.visitEbnf( p_context.ebnf() );
219 
220 
221  return null;
222  }
223 
224  @Override
225  public final IGrammarElement visitEbnf( final ANTLRv4Parser.EbnfContext p_context )
226  {
227  return CCommon.cardinality(

229  ? p_context.blockSuffix().getText()
230  : "",

232  );

258  }
259 
260  @Override
261  public final IGrammarElement visitRuleref( final ANTLRv4Parser.RulerefContext p_context )

264  }
265 
266  @Override
267  public final IGrammarElement visitId( final ANTLRv4Parser.IdContext p_context )
