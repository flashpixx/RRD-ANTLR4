201  i.toAbsolutePath().toString()

208  .toString(),

217  // convert to lower-case

219  // remove package-root name

391  private String label( final String p_line )
392  {
393  final Matcher l_matcher = LANGUAGEMETHODPATTERN.matcher( p_line );
394  if ( !l_matcher.find() )
395  return "";
396 
397  final String[] l_split = l_matcher.group( 0 ).split( "," );
398  final String[] l_return = new String[2];
399 
400  // class name
401  l_return[0] = l_split[0].replace( TRANSLATEMETHODNAME, "" ).replace( "(", "" ).trim();
402  // label name
403  l_return[1] = l_split[1].replace( ")", "" ).replace( "\"", "" ).split( ";" )[0].trim().toLowerCase( Locale.ROOT );
404 
405  return (
406  "this".equals( l_return[0] )
407  ? buildlabel( m_package, m_outerclass, m_outerclass, m_innerclass, l_return[1] )
408  : buildlabel( m_package, m_outerclass, l_return[0].replace( ".class", "" ).replace( m_package + CLASSSEPARATOR, "" ), "", l_return[1] )
409  ).trim().toLowerCase( Locale.ROOT ).replace( CCommon.PACKAGEROOT + CLASSSEPARATOR, "" );
410 
411  }
