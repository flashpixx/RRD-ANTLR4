280  private static Tuple5<File, Set<ETemplate>, Set<File>, Set<String>, Set<String>> initialize(
281  final String p_locale, final String p_outputdirectory, final String[] p_templates,
282  final String[] p_imports, final String[] p_excludes, final String[] p_docclean
283  )
284  {
285  CCommon.language(
286  ( p_locale == null ) || ( p_locale.isEmpty() )
287  ? Locale.getDefault()
288  : Locale.forLanguageTag( p_locale )
289  );
290 
291  return new Tuple5<>(
292  new File( p_outputdirectory ),
293 
294  ( p_templates == null ) || ( p_templates.length == 0 )
295  ? Collections.<ETemplate>emptySet()
296  : Collections.unmodifiableSet(
297  Arrays.stream( p_templates )
298  .map( i -> ETemplate.valueOf( i.trim().toUpperCase() ) )
299  .collect( Collectors.toSet() )
300  ),
301 
302  ( p_imports == null ) || ( p_imports.length == 0 )
303  ? Collections.<File>emptySet()
304  : Collections.unmodifiableSet(
305  Arrays.stream( p_imports )
306  .map( String::trim )
307  .filter( i -> !i.isEmpty() )
308  .map( File::new )
309  .flatMap( i -> CMain.filelist( i, Collections.<File>emptySet(), Collections.<String>emptySet() ) )
310  .collect( Collectors.toSet() )
311  ),
312 
313  ( p_excludes == null ) || ( p_excludes.length == 0 )
314  ? Collections.<String>emptySet()
315  : Collections.unmodifiableSet(
316  Arrays.stream( p_excludes )
317  .map( String::trim )
318  .filter( i -> !i.isEmpty() )
319  .collect( Collectors.toSet() )
320  ),
321 
322  ( p_docclean == null ) || ( p_docclean.length == 0 )
323  ? Collections.<String>emptySet()
324  : Collections.unmodifiableSet(
325  Arrays.stream( p_docclean )
326  .flatMap( i ->
327  {
328  try
329  {
330  return FileUtils.readLines( new File( i ), Charset.defaultCharset() ).stream();
331  }
332  catch ( final IOException l_exception )
333  {
334  return Stream.of( "" );
335  }
336  } )
337  .map( String::trim )
338  .filter( i -> !i.isEmpty() )
339  .collect( Collectors.toSet() )
340  )
341  );
342  }

353  private static Stream<File> filelist( final File p_input, final Set<File> p_import, final Set<String> p_exclude )
354  {
355  if ( !p_input.exists() )
356  throw new RuntimeException( CCommon.languagestring( CMain.class, "notexist", p_input ) );
357 
358  try
359  {
360  return (
361  p_input.isFile()
362  ? Stream.of( p_input )
363  : Files.find( p_input.toPath(), Integer.MAX_VALUE, ( i, j ) -> ( j.isRegularFile() ) && ( !j.isSymbolicLink() ) ).map( Path::toFile )
364  )
365  .filter( i -> i.getName().endsWith( GRAMMARFILEEXTENSION ) )
366  .filter( i -> !p_import.contains( i ) )
367  .filter( i -> !p_exclude.contains( i.getName() ) );
368  }
369  catch ( final IOException l_exception )
370  {
371  throw new RuntimeException( l_exception );
372  }
373  }
