69  @Override
70  public Token emit()
71  {
72  if ( _type == ANTLRv4Lexer.ID )
73  {
74  String firstChar = _input.getText( Interval.of( _tokenStartCharIndex, _tokenStartCharIndex ) );
75  if ( Character.isUpperCase( firstChar.charAt( 0 ) ) )
76  {
77  _type = ANTLRv4Lexer.TOKEN_REF;
78  }
79  else
80  {
81  _type = ANTLRv4Lexer.RULE_REF;
82  }
83 
84  if ( _currentRuleType == Token.INVALID_TYPE )
85  { // if outside of rule def
86  _currentRuleType = _type; // set to inside lexer or parser rule
87  }
88  }
89  else if ( _type == ANTLRv4Lexer.SEMI )
90  { // exit rule def
91  _currentRuleType = Token.INVALID_TYPE;
92  }
93 
94  return super.emit();
95  }
