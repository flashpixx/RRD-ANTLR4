115  private Collection<String> parse( final File p_grammar, final Set<String> p_docuclean, final Map<String, File> p_imports, final ITemplate p_template )
116  throws IOException
117  {
118  // lexing and parsing the input grammar file
119  final CASTVisitorAntLR l_visitor = new CASTVisitorAntLR( p_template, p_docuclean );
120  l_visitor.visit(
121  new ANTLRv4Parser(
122  new CommonTokenStream(
123  new ANTLRv4Lexer(
124  new ANTLRInputStream( new FileInputStream( p_grammar ) )
125  )
126  )
127  ).grammarSpec()
128  );
129 
130  return l_visitor.getGrammarImports().stream()
131  .map( i -> p_imports.get( i.get() ) )
132  .filter( Objects::nonNull )
133  .flatMap( i ->
134  {
135  try
136  {
137  return this.parse( i, p_docuclean, p_imports, p_template ).stream();
138  }
139  catch ( final IOException l_exception )
140  {
141  return Stream.of( l_exception.getMessage() );
