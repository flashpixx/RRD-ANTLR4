120  protected final void replace( final File p_file, final String... p_replacepair ) throws IOException
121  {
122  if ( ( p_replacepair == null ) || ( p_replacepair.length % 2 != 0 ) )
123  throw new IllegalArgumentException( CCommon.languagestring( IBaseTemplate.class, "replaceerror", p_file ) );
124 
125  final CStringReplace l_content = new CStringReplace( FileUtils.readFileToString( p_file, Charset.forName( "UTF-8" ) ) );
126  SequenceM.rangeLong( 0, p_replacepair.length )
127  .sliding( 2, 2 )
128  .forEach( i -> l_content.replaceAll( p_replacepair[i.get( 0 ).intValue()], p_replacepair[i.get( 1 ).intValue()] ) );
129  FileUtils.write( p_file, l_content.get(), Charset.forName( "UTF-8" ) );
130  }

138  @SuppressWarnings( "unchecked" )
139  protected String map( final IGrammarElement p_element )
140  {
141  if ( p_element instanceof IGrammarIdentifier )
142  return this.cardinality( p_element.cardinality(), this.nonterminal( (IGrammarIdentifier) p_element ) );
143 
144  if ( p_element instanceof IGrammarRule )
145  return this.cardinality( p_element.cardinality(), this.rule( (IGrammarRule) p_element ) );
146 
147  if ( p_element instanceof IGrammarSimpleElement<?> )
148  return this.cardinality( p_element.cardinality(), this.terminalvalue( (IGrammarSimpleElement<?>) p_element ) );
149 
150  if ( p_element instanceof IGrammarTerminal )
151  return this.cardinality( p_element.cardinality(), this.terminal( (IGrammarTerminal) p_element ) );
152 
153 
154  if ( p_element instanceof IGrammarNegation )
155  return this.cardinality( p_element.cardinality(), this.negation( ( (IGrammarNegation) p_element ).inner() ) );
156 
157  if ( p_element instanceof IGrammarGroup )
158  return this.cardinality( p_element.cardinality(), this.group( (IGrammarGroup) p_element ) );
159 
160  if ( p_element instanceof IGrammarChoice )
161  return this.cardinality( p_element.cardinality(), this.choice( (IGrammarChoice) p_element ) );
162 
163  if ( p_element instanceof IGrammarSequence )
164  return this.cardinality( p_element.cardinality(), this.sequence( (IGrammarSequence) p_element ) );
165 
166  return "";
167  }
